cmake_minimum_required(VERSION 3.17)


PROJECT(
    PROJETO02
    LANGUAGES CXX
)


add_subdirectory(
    test
)

add_subdirectory(
    proto
)

add_library(
    ${PROJECT_NAME}lib

    
    src/medidor_de_energia/ares.cpp
    src/medidor_de_energia/cronos.cpp
    src/medidor_de_energia/zeus.cpp
    src/medidor_de_energia/apolo.cpp
    src/Application.cpp
    src/EletraEnergySolutions.cpp
    src/medidor_de_energia.cpp
    src/console.cpp
    src/fabrica_de_medidor.cpp
    src/Utility.cpp
    src/Services/Projeto02sevices.cpp
    src/Services/Projeto02Utilityservices.cpp


)

add_executable(
    ${PROJECT_NAME}
  
    src/main.cpp
    
)


target_include_directories(
    ${PROJECT_NAME}lib

    PUBLIC  
        ./include
        ${CMAKE_CURRENT_BINARY_DIR}
 
)

target_compile_features(
    ${PROJECT_NAME}lib
    
    PUBLIC
      cxx_std_20

)

target_link_libraries(
    ${PROJECT_NAME}
    
    PRIVATE
        ${PROJECT_NAME}lib

)

target_link_libraries(
    ${PROJECT_NAME}lib
    
    PUBLIC
        Projeto02Proto
)
