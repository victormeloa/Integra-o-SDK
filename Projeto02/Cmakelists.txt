PROJECT(
    PROJETO02
    LANGUAGES CXX
)


add_subdirectory(
    test
)

add_library(
    ${PROJECT_NAME}lib

    
    src/medidor_de_energia/ares.cpp
    src/medidor_de_energia/cronos.cpp
    src/medidor_de_energia/zeus.cpp
    src/medidor_de_energia/apolo.cpp

    src/EletraEnergySolutions.cpp
    src/medidor_de_energia.cpp
    src/console.cpp
    src/fabrica_de_medidor.cpp
    src/Utility.cpp

    ${PROJECT_NAME}
  
    src/main.cpp
    
)

    ${PROJECT_NAME}lib

    PUBLIC  
    ./include
 
)

    ${PROJECT_NAME}lib
    
    PUBLIC
      cxx_std_20

)

    ${PROJECT_NAME}
    
    PRIVATE
    ${PROJECT_NAME}lib
